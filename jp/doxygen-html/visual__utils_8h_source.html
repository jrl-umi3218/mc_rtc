34 template<typename DataT>
35 rbd::parsers::Visual makeVisual(const DataT & data, const mc_rtc::gui::Color & color)
36 {
37  using GeometryT = rbd::parsers::Geometry;
38  rbd::parsers::Visual out;
39  out.origin = sva::PTransformd::Identity();
40  out.geometry.type = []() {
41  if constexpr(std::is_same_v<DataT, GeometryT::Box>)
42  {
43  return GeometryT::Type::BOX;
44  }
45  else if constexpr(std::is_same_v<DataT, GeometryT::Cylinder>)
46  {
47  return GeometryT::Type::CYLINDER;
48  }
49  else if constexpr(std::is_same_v<DataT, GeometryT::Sphere>)
50  {
51  return GeometryT::Type::SPHERE;
52  }
53  else if constexpr(std::is_same_v<DataT, GeometryT::Mesh>)
54  {
55  return GeometryT::Type::MESH;
56  }
57  else if constexpr(std::is_same_v<DataT, GeometryT::Superellipsoid>)
58  {
59  return GeometryT::Type::SUPERELLIPSOID;
60  }
61  else
62  {
63  static_assert(!std::is_same_v<DataT, DataT>);
64  }
65  }();
66  out.geometry.data = data;
67  setVisualColor(out, color);
68  return out;
69 }

73 auto & getVisualGeometry(rbd::parsers::Visual & visual)

77  mc_rtc::log::error_and_throw("Visual type not matching the expected type");

85  else if constexpr(type == Type::CYLINDER)

88  }
89  else if constexpr(type == Type::SPHERE)

92  }

95  return boost::get<rbd::parsers::Geometry::Mesh>(data);

99  return boost::get<rbd::parsers::Geometry::Superellipsoid>(data);
100  }
101  else
