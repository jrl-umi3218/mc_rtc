107 template<typename PolyhedronT, typename GetColorT>
108 struct ColoredPolyhedronImpl : public PolyhedronT
109 {
110  ColoredPolyhedronImpl(PolyhedronT && poly, GetColorT get_color_fn) : PolyhedronT(poly), get_color_fn_(get_color_fn)
111  {
112  if constexpr(PolyhedronT::type == Elements::PolyhedronTrianglesList)
113  {
114  static_assert(
115  details::CheckReturnType<GetColorT, std::vector<std::array<mc_rtc::gui::Color, 3>>>::value,
116  "Polyhedron color callback must return an std::vector<std::array<mc_rtc::gui::Color, 3>> (color for "
117  "each of the triangle vertices)");
118  }
119  else
120  {
121  static_assert(details::CheckReturnType<GetColorT, std::vector<mc_rtc::gui::Color>>::value,
122  "Polyhedron color callback must return an std::vector<mc_rtc::gui::Color> (color for "
123  "each of the vertices)");
124  }
125  }
126 
128  ColoredPolyhedronImpl() {}
129 
130  static constexpr size_t write_size()
131  {
132  return PolyhedronT::write_size() + 1;
133  }
134 
135  void write(mc_rtc::MessagePackBuilder & builder)
136  {
137  PolyhedronT::write(builder);
138  const auto & colors = get_color_fn_();
139  builder.start_array(colors.size());
140  for(const auto & c : colors)
141  {
142  if constexpr(PolyhedronT::type == Elements::PolyhedronTrianglesList)
143  {
144  builder.start_array(3);
145  c[0].write(builder);
146  c[1].write(builder);
147  c[2].write(builder);
148  builder.finish_array();
149  }
150  else
151  {
152  c.write(builder);
153  }
154  }
155  builder.finish_array();
156  }
157 
158 protected:
159  GetColorT get_color_fn_;
160 };
