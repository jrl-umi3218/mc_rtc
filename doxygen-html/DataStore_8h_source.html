463  template<typename RetT, typename... FuncArgsT, typename... ArgsT>
464  RetT safe_call(const std::string & name, ArgsT &&... args) const
465  {
466  const auto & data = get_data(name);
467  using fn_t = std::function<RetT(FuncArgsT...)>;
468  if(!data.same(typeid(fn_t).hash_code()) && !data.same_name(type_name<fn_t>()))
469  {
470  log::error_and_throw<std::runtime_error>("[{}] Function for key \"{}\" does not have the same signature as the "
471  "requested one. Stored {} but requested {}",
472  name_, name, data.type(), type_name<fn_t>());
473  }
474  auto & fn = *(reinterpret_cast<fn_t *>(data.buffer.get()));
475  return fn(std::forward<ArgsT>(args)...);
