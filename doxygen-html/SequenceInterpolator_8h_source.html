136  Value compute(double currTime)
137  {
138  if(values_.empty())
139  {
140  mc_rtc::log::error_and_throw("SequenceInterpolator requires at least one value");
141  }
142 
143  // Check for out-of-bound access
144  if(currTime >= values_.back().first)
145  {
146  return values_.back().second;
147  }
148  else if(currTime <= values_.front().first)
149  {
150  return values_.front().second;
151  }
152 
153  /*
154  * Efficiently update interval index and associated cached values:
155  * 1/ Check whether currTime is in the current interval
156  * 2/ Check whether currTime is in the next interval
157  * 3/ Perform a binary search
158  * Note that we don't need to check indices bounds here as they are valid by
159  * construction.
160  */
161  auto updateIndex = [this, currTime]() {
162  if(values_[prevIndex_].first < currTime && values_[nextIndex_].first >= currTime)
163  {
164  return;
165  }
166  else if(values_[nextIndex_].first < currTime && values_[nextIndex_ + 1].first >= currTime)
167  {
168  ++prevIndex_;
169  ++nextIndex_;
170  intervalDuration_ = values_[nextIndex_].first - values_[prevIndex_].first;
171  }
172  else
173  {
174  // first element in values_ with time greater or equal to currTime
175  auto nextIt = std::lower_bound(std::begin(values_), std::end(values_), currTime,
176  [](const TimedValue & lhs, const double & rhs) { return lhs.first < rhs; });
177  nextIndex_ = static_cast<size_t>(nextIt - values_.begin());
178  prevIndex_ = nextIndex_ - 1;
179  intervalDuration_ = values_[nextIndex_].first - values_[prevIndex_].first;
180  }
181  };
182 
183  updateIndex();
184  const auto & prevTime = values_[prevIndex_].first;
185  return interpolator_(values_[prevIndex_].second, values_[nextIndex_].second,
186  (currTime - prevTime) / intervalDuration_);
187  }
