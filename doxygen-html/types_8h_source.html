189  style = LineStyle::Dotted;
190  }
191  }
192  }
193

195  {
196  mc_rtc::Configuration config;
197  config.add("color", color);
198  config.add("width", width);
199  if(style == LineStyle::Solid)
200  {
201  config.add("style", "solid");
202  }
203  else if(style == LineStyle::Dotted)
204  {
205  config.add("style", "dotted");
206  }
207  return config;
208  }

213  }
214
