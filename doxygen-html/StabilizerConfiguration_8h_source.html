262  void load(const mc_rtc::Configuration & config)
263  {
264  config("safety_tresholds", safetyThresholds);
265 
266  config("fdqp_weights", fdqpWeights);
267  config("leftFootSurface", leftFootSurface);
268  config("rightFootSurface", rightFootSurface);
269  config("torsoBodyName", torsoBodyName);
270  config("friction", friction);
271 
272  if(config.has("admittance"))
273  {
274  auto admittance = config("admittance");
275  admittance("cop", copAdmittance);
276  admittance("maxVel", copMaxVel);
277  admittance("velFilterGain", mc_filter::utils::clamp(copVelFilterGain, 0, 1));
278  admittance("dfz", dfzAdmittance);
279  admittance("dfz_damping", dfzDamping);
280  }
281  if(config.has("dcm_tracking"))
282  {
283  auto dcmTracking = config("dcm_tracking");
284  if(dcmTracking.has("gains"))
285  {
286  dcmTracking("gains")("prop", dcmPropGain);
287  dcmTracking("gains")("integral", dcmIntegralGain);
288  dcmTracking("gains")("deriv", dcmDerivGain);
289  }
290  dcmTracking("derivator_time_constant", dcmDerivatorTimeConstant);
291  dcmTracking("integrator_time_constant", dcmIntegratorTimeConstant);
292  }
293  config("dcm_bias", dcmBias);
294  if(config.has("tasks"))
295  {
296  auto tasks = config("tasks");
297  if(tasks.has("com"))
298  {
299  tasks("com")("active_joints", comActiveJoints);
300  tasks("com")("stiffness", comStiffness);
301  tasks("com")("weight", comWeight);
302  tasks("com")("height", comHeight);
303  }
304 
305  if(tasks.has("torso"))
306  {
307 
308  tasks("torso")("pitch", torsoPitch);
309  tasks("torso")("stiffness", torsoStiffness);
310  tasks("torso")("weight", torsoWeight);
311  }
312 
313  if(tasks.has("pelvis"))
314  {
315  tasks("pelvis")("stiffness", pelvisStiffness);
316  tasks("pelvis")("weight", pelvisWeight);
317  }
318 
319  if(tasks.has("contact"))
320  {
321  if(tasks("contact").has("damping"))
322  {
323  double d = tasks("contact")("damping");
324  contactDamping = sva::MotionVecd({d, d, d}, {d, d, d});
325  }
326  if(tasks("contact").has("stiffness"))
327  {
328  double k = tasks("contact")("stiffness");
329  contactStiffness = sva::MotionVecd({k, k, k}, {k, k, k});
330  }
331  tasks("contact")("stiffness", contactStiffness);
332  tasks("contact")("weight", contactWeight);
333  }
334  }
335  if(config.has("vdc"))
336  {
337  auto vdc = config("vdc");
338  vdc("frequency", vdcFrequency);
339  vdc("stiffness", vdcStiffness);
340  }
341 
342  config("zmpcc", zmpcc);
343  }
