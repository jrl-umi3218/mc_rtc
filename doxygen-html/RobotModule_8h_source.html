736 inline bool operator==(const RobotModule::Gripper & lhs, const RobotModule::Gripper & rhs)
737 {
738  auto compareMimics = [&]() {
739  auto lmimics = lhs.mimics();
740  auto rmimics = rhs.mimics();
741  if(lmimics == nullptr && rmimics == nullptr)
742  {
743  return true;
744  }
745  if(lmimics == nullptr || rmimics == nullptr)
746  {
747  return false;
748  }
749  return *lmimics == *rmimics;
750  };
751  auto compareSafety = [&]() {
752  auto lsafety = lhs.safety();
753  auto rsafety = rhs.safety();
754  if(lsafety == nullptr && rsafety == nullptr)
755  {
756  return true;
757  }
758  if(lsafety == nullptr || rsafety == nullptr)
759  {
760  return false;
761  }
762  return *lsafety == *rsafety;
763  };
764  return lhs.name == rhs.name && lhs.joints == rhs.joints && lhs.reverse_limits == rhs.reverse_limits && compareSafety()
765  && compareMimics();
766 }
