700 inline bool operator==(const RobotModule::Gripper & lhs, const RobotModule::Gripper & rhs)
701 {
702  auto compareMimics = [&]()
703  {
704  auto lmimics = lhs.mimics();
705  auto rmimics = rhs.mimics();
706  if(lmimics == nullptr && rmimics == nullptr) { return true; }
707  if(lmimics == nullptr || rmimics == nullptr) { return false; }
708  return *lmimics == *rmimics;
709  };
710  auto compareSafety = [&]()
711  {
712  auto lsafety = lhs.safety();
713  auto rsafety = rhs.safety();
714  if(lsafety == nullptr && rsafety == nullptr) { return true; }
715  if(lsafety == nullptr || rsafety == nullptr) { return false; }
716  return *lsafety == *rsafety;
717  };
718  return lhs.name == rhs.name && lhs.joints == rhs.joints && lhs.reverse_limits == rhs.reverse_limits && compareSafety()
719  && compareMimics();
720 }
