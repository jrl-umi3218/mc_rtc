29  template<typename... Args>
30  FormImpl(const std::string & name, Callback cb, Args &&... args) : CallbackElement<Element, Callback>(name, cb)
31  {
32  mc_rtc::MessagePackBuilder builder(data_);
33  count_ = sizeof...(Args);
34  write_elements(builder, std::forward<Args>(args)...);
35  data_size_ = builder.finish();
36  }
37 
38  template<typename T>
39  void addElement(T && element)
40  {
41  count_ += 1;
42  using ElementT = typename std::decay<T>::type;
43  if(ElementT::is_dynamic())
44  {
45  addDynamicElement(std::forward<T>(element));
46  }
47  else
48  {
49  std::vector<char> data = data_;
50  mc_rtc::MessagePackBuilder builder(data_);
51  builder.write_object(data.data(), data_size_);
52  builder.start_array(element.write_size());
53  element.write(builder);
54  builder.finish_array();
55  data_size_ = builder.finish();
56  }
57  }
58 
59  static constexpr size_t write_size()
60  {
61  return CallbackElement<Element, Callback>::write_size() + 1;
62  }
63 
64  void write(mc_rtc::MessagePackBuilder & builder)
65  {
66  CallbackElement<Element, Callback>::write(builder);
67  builder.start_array(count_);
68  for(const auto & el : dynamic_elements_)
69  {
70  el(builder);
71  }
72  builder.write_object(data_.data(), data_size_);
73  for(size_t i = dynamic_elements_.size() + 1; i < count_; ++i)
74  {
75  builder.write_object(nullptr, 0);
76  }
77  builder.finish_array();
78  }
79 
81  FormImpl() {}
82 
83 private:
84  template<typename... Args>
85  void write_elements(mc_rtc::MessagePackBuilder &, Args &&...)
86  {
87  }
88 
89  template<typename Arg, typename... Args>
90  void write_elements(mc_rtc::MessagePackBuilder & builder, Arg && element, Args &&... args)
91  {
92  using ElementT = typename std::decay<Arg>::type;
93  if(ElementT::is_dynamic())
94  {
95  addDynamicElement(std::forward<Arg>(element));
96  }
97  else
98  {
99  builder.start_array(element.write_size());
100  element.write(builder);
101  builder.finish_array();
102  }
103  write_elements(builder, std::forward<Args>(args)...);
104  }
105 
106  template<typename T>
107  void addDynamicElement(T && element)
108  {
109  auto callback = [element](mc_rtc::MessagePackBuilder & builder) mutable {
110  builder.start_array(element.write_size());
111  element.write(builder);
112  builder.finish_array();
113  };
114  dynamic_elements_.push_back(callback);
115  }
116 
117  size_t count_;
118  std::vector<std::function<void(mc_rtc::MessagePackBuilder &)>> dynamic_elements_;
119  std::vector<char> data_;
120  size_t data_size_;
121 };
